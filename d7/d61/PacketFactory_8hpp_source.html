39 static void addCommonsToPacketByEntity(Protocol::Packet<Id> &packet, Engine::ECS::Registry &registry,
40 Engine::ECS::Entity entity)
41 {
42 auto child = registry.getComponents<Engine::ECS::Components::Common::Child>()[entity];
43
44 if (child.has_value())
45 {
46 packet << Protocol::ComponentId::CHILD;
47 packet.injectString(child->name);
48 }
49
50 auto evolve = registry.getComponents<Engine::ECS::Components::Common::Evolve>()[entity];
51
52 if (evolve.has_value())
53 {
54 packet << Protocol::ComponentId::EVOLVE;
55 packet.injectString(evolve->name);
56 }
57
58 auto health = registry.getComponents<Engine::ECS::Components::Common::Health>()[entity];
59
60 if (health.has_value())
61 {
62 packet << Protocol::ComponentId::HEALTH;
63 packet << health->currentHealth;
64 packet << health->maxHealth;
65 packet << health->shield;
66 packet << health->maxShield;
67 }
68
69 auto id = registry.getComponents<Engine::ECS::Components::Common::Id>()[entity];
70
71 if (id.has_value())
72 {
73 packet << Protocol::ComponentId::ID;
74 packet << id->id;
75 }
76
77 auto level = registry.getComponents<Engine::ECS::Components::Common::Level>()[entity];
78
79 if (level.has_value())
80 {
81 packet << Protocol::ComponentId::LEVEL;
82 packet << level->level;
83 packet.injectString(level->currentWeapon);
84 packet << level->currentExp;
85 packet << level->requiredExp;
86 }
87
88 auto parent = registry.getComponents<Engine::ECS::Components::Common::Parent>()[entity];
89
90 if (parent.has_value())
91 {
92 packet << Protocol::ComponentId::PARENT;
93 packet << parent->entity;
94 }
95
96 auto tag = registry.getComponents<Engine::ECS::Components::Common::Tag>()[entity];
97
98 if (tag.has_value())
99 {
100 packet << Protocol::ComponentId::TAG;
101 packet.injectString(tag->tag);
102 }
103
104 auto template_ = registry.getComponents<Engine::ECS::Components::Common::Template>()[entity];
105
106 if (template_.has_value())
107 {
108 packet << Protocol::ComponentId::TEMPLATE;
109 packet.injectString(template_->name);
110 }
111
112 auto timer = registry.getComponents<Engine::ECS::Components::Common::Timer>()[entity];
113
114 if (timer.has_value())
115 {
116 packet << Protocol::ComponentId::TIMER;
117 packet << timer->lastTime.time_since_epoch().count();
118 packet << timer->maxTime;
119 }
120
121 auto weapon = registry.getComponents<Engine::ECS::Components::Common::Weapon>()[entity];
122
123 if (weapon.has_value())
124 {
125 packet << Protocol::ComponentId::WEAPON;
126 packet << weapon->minDamage;
127 packet << weapon->maxDamage;
128 packet << weapon->chargeMaxTime;
129 packet << weapon->fireRate;
130 packet << weapon->level;
131 }

143 static void add2dToPacketByEntity(Packet<Id> &packet, Engine::ECS::Registry &registry, Engine::ECS::Entity entity)
144 {
145 auto transform = registry.getComponents<Engine::ECS::Components::_2D::Transform>()[entity];
146
147 if (transform.has_value())
148 {
149 packet << ComponentId::TRANSFORM;
150 packet << transform->_position.vec.x;
151 packet << transform->_position.vec.y;
152 packet << transform->_rotation;
153 packet << transform->_scale.vec.x;
154 packet << transform->_scale.vec.y;
155 }
156
157 auto movable = registry.getComponents<Engine::ECS::Components::_2D::Movable>()[entity];
158
159 if (movable.has_value())
160 {
161 packet << ComponentId::MOVABLE;
162 packet << movable->_velocity.vec.x;
163 packet << movable->_velocity.vec.y;
164 packet << movable->_acceleration.vec.x;
165 packet << movable->_acceleration.vec.y;
166 }
167
168 auto control = registry.getComponents<Engine::ECS::Components::_2D::Control>()[entity];
169
170 if (control.has_value())
171 {
172 packet << ComponentId::CONTROL;
173 packet << control->_up;
174 packet << control->_down;
175 packet << control->_left;
176 packet << control->_right;
177 packet << control->_shoot;
178 }
179
180 auto collider = registry.getComponents<Engine::ECS::Components::_2D::Collider>()[entity];
181
182 if (collider.has_value())
183 {
184 packet << ComponentId::COLLIDER;
185 packet << collider->_size.vec.x;
186 packet << collider->_size.vec.y;
187 }
188
189 auto rigidbody = registry.getComponents<Engine::ECS::Components::_2D::RigidBody>()[entity];
190

193 packet << ComponentId::RIGIDBODY;
