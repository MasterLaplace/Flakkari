104 template <typename DataType> friend Packet<Id> &operator>>(Packet<Id> &packet, DataType &data)

154 [[nodiscard]] bool deserialize(const Network::Buffer &buffer)
155 {
156 if (buffer.size() < header.size())
157 return FLAKKARI_LOG_WARNING("Buffer is too small to deserialize a packet."), false;
158 std::memcpy(&header, buffer.data(), header.size());
159 if (header._priority >= Priority::MAX_PRIORITY)
160 return FLAKKARI_LOG_WARNING("Priority is too big (" + std::to_string((int) header._priority) + ")"), false;
161 if (header._apiVersion >= ApiVersion::MAX_VERSION)
162 return FLAKKARI_LOG_WARNING("ApiVersion is too big (" + std::to_string((int) header._apiVersion) + ")"),
163 false;
164 if (header._commandId >= CommandId::MAX_COMMAND_ID)
165 return FLAKKARI_LOG_WARNING("CommandId is too big (" + std::to_string((int) header._commandId) + ")"),
166 false;
167 if (header._contentLength > buffer.size() - header.size())
168 return false;
169 payload = buffer.extractData(header.size(), header._contentLength);
170 return true;
171 }
