303 void rotate(const Type &angleRadians)
304 {
305 Type c = cos(angleRadians);
306 Type s = sin(angleRadians);
307 Type x = v[0] * c - v[1] * s;
308 Type y = v[0] * s + v[1] * c;
309 v[0] = x;
310 v[1] = y;
311 }

313 void quaternionRotate(Type x, Type y, Type z, Type w)
314 {
315 v[0] = v[3] * x + v[0] * w + v[1] * z - v[2] * y;
316 v[1] = v[3] * y - v[0] * z + v[1] * w + v[2] * x;
317 v[2] = v[3] * z + v[0] * y - v[1] * x + v[2] * w;
318 v[3] = v[3] * w - v[0] * x - v[1] * y - v[2] * z;
319 }

327 void rotate(const Vector<Type> &axis, float angleDegrees)
328 {
329 const double _pi = 3.14159265358979323846;
330 double angleRadians = (angleDegrees * _pi / 180) / 2.0;
331 double sinHalfAngle = sin(angleRadians);
332 auto normalizedAxis = axis.normalized();
333 quaternionRotate(normalizedAxis.v[0] * sinHalfAngle, normalizedAxis.v[1] * sinHalfAngle,
334 normalizedAxis.v[2] * sinHalfAngle, cos(angleRadians));
335 }
